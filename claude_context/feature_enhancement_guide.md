# 기존 함수 수정 대신 래퍼 함수 생성
def enhanced_analyze(data, use_v5=False):
    if use_v5:
        return new_v5_analysis(data)
    else:
        return original_analyze(data)  # 기존 함수 보존

# 기존 클래스 상속으로 확장
class EnhancedAnalyzer(OriginalAnalyzer):
    def __init__(self):
        super().__init__()
        self.v5_features = V5Features()
    
    def analyze_with_v5(self, data):
        base_result = super().analyze(data)
        enhanced_result = self.v5_features.enhance(base_result)
        return enhanced_result
```

def validate_new_feature_input(data):
    """새 기능 입력 검증 템플릿"""
    # 1. 필수 필드 확인
    required_fields = ['field1', 'field2']
    for field in required_fields:
        if field not in data:
            raise ValueError(f"필수 필드 누락: {field}")
    
    # 2. 데이터 타입 검증
    if not isinstance(data.get('numeric_field'), (int, float)):
        raise ValueError("숫자 타입이 아님")
    
    # 3. 범위 검증
    if not 0 <= data.get('score', 0) <= 100:
        raise ValueError("점수는 0-100 범위여야 함")
    
    # 4. SQL 인젝션 방지
    data['text_field'] = sanitize_input(data.get('text_field', ''))
    
    return data
```

def check_permission(user_role, required_permission):
